## 一、设计规范

### 1.1 命名规范

| 规范 | 说明 |
|------|------|
| 表名 | 小写字母 + 下划线，如 `user_account` |
| 主键 | `id` (UUID) |
| 时间戳 | `created_at`, `updated_at` ( timestamptz ) |
| 软删除 | `deleted_at` ( timestamptz, nullable ) |
| 索引 | `idx_<表名>_<字段>` |
| 外键 | `fk_<表名>_<参照表>` |

### 1.2 字段类型规范

| 类型 | 说明 |
|------|------|
| UUID | 主键使用 uuid，主键使用 gen_random_uuid() |
| timestamptz | 所有时间戳字段使用 timestamptz，默认 CURRENT_TIMESTAMP |
| jsonb | 扩展数据使用 jsonb |
| decimal | 金额使用 decimal(10,2) |
| 字符串 | varchar(255) 为默认，text 用于长文本 |

### 1.3 通用字段

所有表都包含以下字段：

```sql
id          uuid        PRIMARY KEY DEFAULT gen_random_uuid()
created_at  timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP
updated_at  timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP
deleted_at  timestamptz
```

## 二、用户与认证模块

### 2.1 用户表 (users)

```sql
CREATE TABLE users (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    phone varchar(20) UNIQUE,
    email varchar(255) UNIQUE,
    password_hash varchar(255),
    nickname varchar(100),
    avatar varchar(500),
    gender varchar(20) DEFAULT 'unknown',  -- unknown/male/female
    birthday date,
    status varchar(20) DEFAULT 'active',   -- active/inactive/suspended/destroyed
    last_login_at timestamptz,
    last_login_device varchar(255),
    created_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,
    deleted_at timestamptz
);

CREATE INDEX idx_users_phone ON users(phone) WHERE deleted_at IS NULL;
CREATE INDEX idx_users_status ON users(status) WHERE deleted_at IS NULL;
```

### 2.2 设备表 (user_devices)

```sql
CREATE TABLE user_devices (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id uuid NOT NULL REFERENCES users(id),
    device_id varchar(255) NOT NULL,
    device_name varchar(255),
    device_type varchar(50),  -- ios/android/h5/mini
    os_version varchar(50),
    app_version varchar(50),
    last_active_at timestamptz,
    is_current boolean DEFAULT false,
    status varchar(20) DEFAULT 'active',
    created_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,
    deleted_at timestamptz,
    
    CONSTRAINT fk_user_devices_user FOREIGN KEY (user_id) REFERENCES users(id),
    UNIQUE(user_id, device_id)
);

CREATE INDEX idx_user_devices_user ON user_devices(user_id) WHERE deleted_at IS NULL;
```

### 2.3 认证令牌表 (auth_tokens)

```sql
CREATE TABLE auth_tokens (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id uuid NOT NULL,
    token_hash varchar(255) NOT NULL,
    token_type varchar(20) DEFAULT 'access',  -- access/refresh
    device_id varchar(255),
    expires_at timestamptz NOT NULL,
    last_used_at timestamptz,
    created_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,
    deleted_at timestamptz,
    
    CONSTRAINT fk_auth_tokens_user FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE INDEX idx_auth_tokens_token ON auth_tokens(token_hash) WHERE deleted_at IS NULL;
CREATE INDEX idx_auth_tokens_user ON auth_tokens(user_id) WHERE deleted_at IS NULL;
CREATE INDEX idx_auth_tokens_expires ON auth_tokens(expires_at) WHERE deleted_at IS NULL;
```

### 2.4 微信绑定表 (wechat_bindings)

```sql
CREATE TABLE wechat_bindings (
```

### 2.5 验证码表 (verification_codes)

```sql
CREATE TABLE verification_codes (
```

## 三、生命管理模块

### 3.1 生命表 (lives) - 核心表

```sql
CREATE TABLE lives (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id uuid NOT NULL REFERENCES users(id),
    family_id uuid REFERENCES families(id),
    
    life_type varchar(20) NOT NULL,  -- fetus/baby/child
    nickname varchar(100),
    avatar varchar(500),
    gender varchar(20) DEFAULT 'unknown',
    
    birth_date date,
    due_date date,
    gestation_week int,
    
    current_stage varchar(20) NOT NULL,  -- fetus/infant/toddler/child/teen/adult/elder
    stage_start_date date NOT NULL,
    
    mother_name varchar(100),
    birth_weight decimal(6,2),  -- kg
    birth_height decimal(6,2),   -- cm
    birth_hospital varchar(255),
    
    is_active boolean DEFAULT true,
    status varchar(20) DEFAULT 'active',
    
    created_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,
    deleted_at timestamptz,
    
    CONSTRAINT fk_lives_user FOREIGN KEY (user_id) REFERENCES users(id),
    CONSTRAINT chk_lives_stage CHECK (current_stage IN ('fetus', 'infant', 'toddler', 'child', 'teen', 'adult', 'elder'))
);

CREATE INDEX idx_lives_user ON lives(user_id) WHERE deleted_at IS NULL;
CREATE INDEX idx_lives_family ON lives(family_id) WHERE deleted_at IS NULL;
CREATE INDEX idx_lives_current_stage ON lives(current_stage) WHERE deleted_at IS NULL;
CREATE INDEX idx_lives_birth_date ON lives(birth_date) WHERE deleted_at IS NULL;
```

### 3.2 生命阶段历史表 (life_stage_history)

```sql
CREATE TABLE life_stage_history (

    created_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,
    deleted_at timestamptz,
    
    CONSTRAINT fk_stage_history_life FOREIGN KEY (life_id) REFERENCES lives(id)
);

CREATE INDEX idx_stage_history_life ON life_stage_history(life_id) WHERE deleted_at IS NULL;
```

### 3.3 生命元数据表 (life_metadata)

```sql
CREATE TABLE life_metadata (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    life_id uuid NOT NULL REFERENCES lives(id),
    metadata_key varchar(100) NOT NULL,
    metadata_value jsonb,
    
    created_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,

    deleted_at timestamptz,
```

## 五、时间轴模块

### 5.1 时间轴表 (timeline)

```sql
CREATE TABLE timeline (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    life_id uuid NOT NULL REFERENCES lives(id),
    
    event_type varchar(30) NOT NULL,  -- growth/health/education/emotion/family/milestone/custom
    stage_type varchar(20) NOT NULL,   -- fetus/infant/toddler/child/teen/adult/elder
    
    title varchar(255) NOT NULL,
    description text,
    event_date date NOT NULL,
    
    space_type varchar(20) NOT NULL,  -- public/private
    owner_life_id uuid NOT NULL,       -- 实际发布者的life_id
    
    tags text[],
    emotion varchar(20),               -- positive/neutral/negative
    emotion_score decimal(3,2),
    
    source_plugin varchar(50),
    source_content_id uuid,
    
    ai_summary text,
    ai_insights jsonb,
    
    like_count int DEFAULT 0,
    comment_count int DEFAULT 0,
    view_count int DEFAULT 0,
    
    status varchar(20) DEFAULT 'published',  -- draft/published/private/deleted
    published_at timestamptz,
    
    created_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at timestamptz NOT NULL DEFAULT CURRENT_TIMESTAMP,
    deleted_at timestamptz,
    
    CONSTRAINT fk_timeline_life FOREIGN KEY (life_id) REFERENCES lives(id)
);

CREATE INDEX idx_timeline_life ON timeline(life_id) WHERE deleted_at IS NULL;
CREATE INDEX idx_timeline_owner_life ON timeline(owner_life_id) WHERE deleted_at IS NULL;
CREATE INDEX idx_timeline_space_type ON timeline(space_type) WHERE deleted_at IS NULL;
CREATE INDEX idx_timeline_event_type ON timeline(event_type) WHERE deleted_at IS NULL;
CREATE INDEX idx_timeline_stage_type ON timeline(stage_type) WHERE deleted_at IS NULL;
CREATE INDEX idx_timeline_event_date ON timeline(event_date DESC) WHERE deleted_at IS NULL;
CREATE INDEX idx_timeline_space_owner ON timeline(space_type, owner_life_id) WHERE deleted_at IS NULL;

-- TimescaleDB 超表配置（可选，用于时序优化）
SELECT create_hypertable('timeline', 'event_date', if_not_exists => TRUE);
```

## 十五、TimescaleDB 超表配置（可选）

```sql
-- 为时序数据创建超表
SELECT create_hypertable('timeline', 'event_date', if_not_exists => TRUE);
SELECT create_hypertable('ai_messages', 'created_at', if_not_exists => TRUE);
SELECT create_hypertable('notifications', 'created_at', if_not_exists => TRUE);

-- 创建持续聚合用于趋势分析
CREATE MATERIALIZED VIEW diary_emotion_trend
WITH (timescaledb.continuous) AS
SELECT time_bucket('1 day', diary_date) AS bucket,
       life_id,
       AVG(mood_score) AS avg_mood,
       mode() WITHIN GROUP (ORDER BY mood) AS dominant_mood
FROM diaries
WHERE deleted_at IS NULL
GROUP BY bucket, life_id;

-- 创建保留策略
ALTER MATERIALIZED VIEW diary_emotion_trend
SET (
    timescaledb.continuous,
    timescaledb.refresh_interval = '1 hour',
    timescaledb.ignore_invalidation_older_than = '7 days'
);
```
