**根 Makefile 示例**：

```makefile
.PHONY: install dev build test clean

install:
    npm run install:mobile && npm run install:admin
    cd backend && go mod download
    cd ai-service && pip install -r requirements.txt

dev:
    npm run dev:mobile
    make dev:api
    make dev:ai
    make dev:admin

dev:api:
    cd backend && go run cmd/api/main.go

dev:ai:
    cd ai-service && uvicorn app.main:app --reload

dev:admin:
    cd admin && npm run dev

build:mobile:
    cd mobile && npm run build:android

build:api:
    cd backend && go build -o bin/api ./cmd/api

build:ai:
    cd ai-service && docker build -t ai-service .

build:admin:
    cd admin && npm run build

test:
    npm run test:mobile
    cd backend && go test ./...
    cd ai-service && pytest

clean:
    rm -rf mobile/dist
    rm -rf backend/bin
    rm -rf admin/.next
```
